int main() {             // Point d'entrée
    printf("Hello\n");   // Afficher
    return 0;            // Fin OK
}

Contrôle de flux

Conditionnelles & Boucles
// if / else if / else
if (n >= 10) { ... } else if (n >= 5) { ... } else { ... }

// Ternaire : condition ? vrai : faux
int max = (a > b) ? a : b;

// switch — C.C.B.D.
switch(x) { case 1: ...; break; default: ...; }

// for — I.C.I. (Init, Condition, Incrément)
for (int i = 0; i < n; i++) { ... }

// while (0+ tours)           do-while (1+ tours)
while (cond) { ... }         do { ... } while (cond);

// break = sortir, continue = tour suivant

Fonctions — T.N.P.C.R.

Fonctions
// Prototype
int carre(int n);

// Définition
int carre(int n) { return n * n; }

// void = pas de retour
void dire(char msg[]) { printf("%s\n", msg); }

// Par valeur (copie)      Par adresse (pointeur)
void f(int x) { x++; }   void f(int *p) { (*p)++; }
f(a);                      f(&a);

// static : persiste entre appels
static int count = 0;

Tableaux & Chaînes

Arrays & Strings
int t[5] = {1,2,3,4,5};   // Indice 0 à 4 !
int m[3][4];               // Matrice 3×4
char s[20] = "Hello";     // +1 pour '\0'

// string.h — L.C.C.C.
strlen(s)                  // Longueur (sans '\0')
strcpy(dst, src)           // Copier (PAS de = !)
strcat(dst, src)           // Concaténer
strcmp(a, b) == 0          // Comparer (0 = égaux)

// Tableau en paramètre : toujours avec taille
void f(int t[], int n) { ... }

Pointeurs

Pointeurs
int *p = &x;    // & = Adresse, * = Accéder
*p = 42;          // Modifie x via p
int *p = NULL;    // Pointeur sûr

// tab ≡ &tab[0]        tab[i] ≡ *(tab + i)

// Allocation — M.C.R.F.
int *t = (int*)malloc(n * sizeof(int));   // Non init
int *t = (int*)calloc(n, sizeof(int));    // Init à 0
t = (int*)realloc(t, m * sizeof(int));    // Redim
free(t); t = NULL;                         // Libérer !

Structures & Enum

struct, typedef, enum
typedef struct {
    char nom[30]; int age; float note;
} Etudiant;

Etudiant e = {"Alice", 20, 15.5};
e.nom        // accès direct (point)
p->nom       // accès via pointeur (flèche) = (*p).nom

typedef enum { ROUGE, VERT, BLEU } Couleur;

Fichiers — R.W.A.

Fichiers
FILE *f = fopen("f.txt", "r");  // r=lire, w=écrire, a=ajouter
if (f == NULL) return 1;        // Toujours vérifier !

fprintf(f, "%d", x);            // Écrire formaté
fgets(buf, sizeof(buf), f);     // Lire une ligne
fscanf(f, "%d", &x);           // Lire formaté
fclose(f);                       // OBLIGATOIRE

// Binaire : fwrite(data, size, count, f) / fread(...)

C++ — L'essentiel

C++ vs C
#include <iostream>
#include <string>
using namespace std;

cout << "Sortie" << endl;    // Remplace printf
cin >> x;                     // Remplace scanf
string s = "Hello" + " World"; // Remplace char[] + strcat
void f(int &n) { n++; }       // Référence (pas de *)
int *p = new int(42);          // Remplace malloc
delete p;                       // Remplace free

class Nom {
private: int attr;
public:
    Nom(int a) : attr(a) {}    // Constructeur
    int get() { return attr; } // Getter — P.A.M.C.
};
class Fille : public Mere { }; // Héritage
